function aStar(grid, start, target) {
  const openSet = [start];
  const closedSet = new Set();
  
  // fScore = gScore + hScore
  start.g = 0;
  start.h = manhattanDistance(start, target);
  start.f = start.g + start.h;

  while (openSet.length > 0) {
    // 1. Get node with lowest f score (Tie-breaker: lowest h score)
    const current = openSet.reduce((min, node) => 
      (node.f < min.f || (node.f === min.f && node.h < min.h)) ? node : min
    );

    if (current === target) return reconstructPath(current);

    openSet.remove(current);
    closedSet.add(current);

    for (const neighbor of getNeighbors(grid, current)) {
      if (closedSet.has(neighbor) || neighbor.isWall) continue;

      const tentativeG = current.g + 1;

      if (!openSet.includes(neighbor)) {
        openSet.push(neighbor);
      } else if (tentativeG >= neighbor.g) {
        continue;
      }

      neighbor.parent = current;
      neighbor.g = tentativeG;
      neighbor.h = manhattanDistance(neighbor, target);
      neighbor.f = neighbor.g + neighbor.h;
    }
  }
}
